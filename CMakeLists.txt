cmake_minimum_required(VERSION 2.6)
project(ecell4)
if(ECELL4_ENABLE_PROFILING)
    add_definitions(-pg)
    set(CMAKE_EXE_LINKER_FLAGS "-pg ${CMAKE_EXE_LINKER_FLAGS}")
    set(CMAKE_SHARED_LINKER_FLAGS "-pg ${CMAKE_SHARED_LINKER_FLAGS}")
    set(CMAKE_MODULE_LINKER_FLAGS "-pg ${CMAKE_MODULE_LINKER_FLAGS}")
endif(ECELL4_ENABLE_PROFILING)
option(WITH_LATTICE "with lattice" ON)
include_directories(${PROJECT_SOURCE_DIR})
include_directories(${PROJECT_BINARY_DIR})
enable_testing()
add_subdirectory(ecell4)
